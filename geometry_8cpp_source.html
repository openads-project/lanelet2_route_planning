15double wrapAngle(const double angle) {
16 double wrapped_angle = angle;
17 while (wrapped_angle > M_PI) wrapped_angle -= 2 * M_PI;
18 while (wrapped_angle < -M_PI) wrapped_angle += 2 * M_PI;
19 return wrapped_angle;
20}

22double angleBetweenVectors(const Eigen::Vector2d& v1, const Eigen::Vector2d& v2) {
23 double dot_product = v1.dot(v2);
24 double determinant = v1.x() * v2.y() - v1.y() * v2.x();
25 double angle = std::atan2(determinant, dot_product);
26 return wrapAngle(angle);
27}

29std::optional<IntersectionOfLinesResult> intersectionOfLines(const std::vector<Eigen::Vector2d>& line1,
30 const std::vector<Eigen::Vector2d>& line2) {
31 IntersectionOfLinesResult result;
32
33 // check if lines are valid
34 if (line1.size() < 2 || line2.size() < 2) {
35 return std::nullopt;
36 }
37
38 // define straight lines (a1 x + b1 y = c1) and (a2 x + b2 y = c2) through line points
39 const double a1 = line1[1].y() - line1[0].y();
40 const double b1 = line1[0].x() - line1[1].x();
41 const double c1 = line1[0].x() * line1[1].y() - line1[1].x() * line1[0].y();
42 const double a2 = line2[1].y() - line2[0].y();
43 const double b2 = line2[0].x() - line2[1].x();
44 const double c2 = line2[0].x() * line2[1].y() - line2[1].x() * line2[0].y();
45
46 // find intersection of both lines by solving for x and y
47 const double det = a1 * b2 - a2 * b1;
48 if (det != 0) {
49 const double x = (b2 * c1 - b1 * c2) / det;
50 const double y = (a1 * c2 - a2 * c1) / det;
51 result.intersection = Eigen::Vector2d(x, y);
52
53 // check if intersection point is within line segments
54 result.intersects_line1 = (x >= std::min(line1[0].x(), line1[1].x()) && x <= std::max(line1[0].x(), line1[1].x()) &&
55 y >= std::min(line1[0].y(), line1[1].y()) && y <= std::max(line1[0].y(), line1[1].y()));
56 result.intersects_line2 = (x >= std::min(line2[0].x(), line2[1].x()) && x <= std::max(line2[0].x(), line2[1].x()) &&
57 y >= std::min(line2[0].y(), line2[1].y()) && y <= std::max(line2[0].y(), line2[1].y()));
58 } else {
59 return std::nullopt;
60 }
61
62 return result;
63}

65Eigen::Vector2d tangentOfPointAlongLineString(const Eigen::Vector2d& point,
66 const Eigen::Vector2d& prev_point,
67 const Eigen::Vector2d& next_point) {
68 Eigen::Vector2d tangent;
69
70 if (point == prev_point && point != next_point) { // single line segment
71 tangent = (next_point - point).normalized();
72 } else if (point != prev_point && point == next_point) { // single line segment
73 tangent = (point - prev_point).normalized();
74 } else if (point == prev_point && point == next_point) { // single point
75 tangent = Eigen::Vector2d(0.0, 0.0);
76 } else { // proper two line segments with previous and next point
77 const Eigen::Vector2d prev_to_point_unit = (point - prev_point).normalized();
78 const Eigen::Vector2d point_to_next_unit = (next_point - point).normalized();
79 tangent = (prev_to_point_unit + point_to_next_unit).normalized();
80 }
81
82 return tangent;
83}

94std::vector<Eigen::Vector3d> resampleLineString(const std::vector<Eigen::Vector3d>& line_string,
95 const double delta,
96 double& offset) {
97 std::vector<Eigen::Vector3d> resampled_line_string;
98
99 // set initial sampling distance
100 double sampling_distance = delta;
101 if (offset == 0.0) { // if no offset, start with first line point
102 resampled_line_string.push_back(line_string.front());
103 } else { // else sample first point with offset != delta
104 sampling_distance = offset;
105 }
106
107 // loop over all line segments
108 for (size_t i = 1; i < line_string.size(); ++i) {
109 // determine 2D segment length and 3D direction
110 double segment_length = (to2d(line_string[i]) - to2d(line_string[i - 1])).norm();
111 const Eigen::Vector3d segment_direction = (line_string[i] - line_string[i - 1]) / segment_length;
112
113 // sample points along segment, increasing sampling_distance by delta
114 while (segment_length >= sampling_distance) {
115 const Eigen::Vector3d resampled_point = line_string[i - 1] + sampling_distance * segment_direction;
116 resampled_line_string.push_back(resampled_point);
117 sampling_distance += delta;
118 }
119
120 // reset sampling_distance for next segment, including overshoot
121 sampling_distance = sampling_distance - segment_length;
122 }
123
124 // save overshoot in outgoing offset parameter
125 offset = sampling_distance;
126
127 return resampled_line_string;
128}

138std::optional<ProjectPointToLineStringAlongAxisResult> projectPointToLineStringAlongAxis(
139 const Eigen::Vector2d& point, const Eigen::Vector2d& axis, const std::vector<Eigen::Vector2d>& line_string) {
140 ProjectPointToLineStringAlongAxisResult result;
141 result.found_intersection_with_line_segment = false;
142 double closest_distance_to_line_segment = std::numeric_limits<double>::max();
143 bool found_at_least_one_intersection = false;
144
145 // define straight line along axis at point
146 std::vector<Eigen::Vector2d> axis_line = {point, point + axis};
147
148 // loop over line segments
149 for (size_t i = 0; i < line_string.size() - 1; ++i) {
150 std::vector<Eigen::Vector2d> line_segment = {line_string[i], line_string[i + 1]};
151
152 // find intersection of axis line and line segment
153 if (auto inner_result = intersectionOfLines(axis_line, line_segment)) {
154 const Eigen::Vector2d& intersection = inner_result->intersection;
155 const bool intersects_line_segment = inner_result->intersects_line2;
156 found_at_least_one_intersection = true;
157 if (intersects_line_segment) {
158 result.found_intersection_with_line_segment = true;
159 result.projected_point = intersection;
160 break;
161 }
162 double distance_to_line_segment =
163 std::min((intersection - line_string[i]).norm(), (intersection - line_string[i + 1]).norm());
164 if (distance_to_line_segment < closest_distance_to_line_segment) {
165 closest_distance_to_line_segment = distance_to_line_segment;
166 result.projected_point = intersection;
167 }
168 }
169 }
170
171 if (!found_at_least_one_intersection) {
172 return std::nullopt;
173 }
174
175 return result;
176}

178std::optional<ProjectPointToLineStringAlongAxisResult> projectPointToLineStringAlongNormal(
179 const Eigen::Vector2d& point,
180 const Eigen::Vector2d& prev_point,
181 const Eigen::Vector2d& next_point,
182 const std::vector<Eigen::Vector2d>& line_string) {
183 // find normal to tangent at point
184 Eigen::Vector2d normal = normalOfPointAlongLineString(point, prev_point, next_point);
185 if (normal == Eigen::Vector2d(0.0, 0.0)) {
186 return std::nullopt;
187 }
188
189 // project current point to other line along normal to tangent
190 if (auto result = projectPointToLineStringAlongAxis(point, normal, line_string)) {
191 return result;
192 }
193 return std::nullopt;
194}
