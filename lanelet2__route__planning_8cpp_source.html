78void Lanelet2RoutePlanning::declareAndLoadParameter(const std::string& name,
79 T& param,
80 const std::string& description,
81 const bool add_to_auto_reconfigurable_params,
82 const bool is_required,
83 const bool read_only,
84 const std::optional<double>& from_value,
85 const std::optional<double>& to_value,
86 const std::optional<double>& step_value,
87 const std::string& additional_constraints) {
88 rcl_interfaces::msg::ParameterDescriptor param_desc;
89 param_desc.description = description;
90 param_desc.additional_constraints = additional_constraints;
91 param_desc.read_only = read_only;
92
93 auto type = rclcpp::ParameterValue(param).get_type();
94
95 if (from_value.has_value() && to_value.has_value()) {
96 if constexpr (std::is_integral_v<T>) {
97 rcl_interfaces::msg::IntegerRange range;
98 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
99 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
100 param_desc.integer_range = {range};
101 } else if constexpr (std::is_floating_point_v<T>) {
102 rcl_interfaces::msg::FloatingPointRange range;
103 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
104 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
105 param_desc.floating_point_range = {range};
106 } else {
107 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range", name.c_str());
108 }
109 }
110
111 this->declare_parameter(name, type, param_desc);
112
113 try {
114 param = this->get_parameter(name).get_value<T>();
115 std::stringstream ss;
116 ss << "Loaded parameter '" << name << "': ";
117 if constexpr (is_vector_v<T>) {
118 ss << "[";
119 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
120 ss << "]";
121 } else {
122 ss << param;
123 }
124 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
125 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
126 if (is_required) {
127 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");
128 exit(EXIT_FAILURE);
129 } else {
130 std::stringstream ss;
131 ss << "Missing parameter '" << name << "', using default value: ";
132 if constexpr (is_vector_v<T>) {
133 ss << "[";
134 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
135 ss << "]";
136 } else {
137 ss << param;
138 }
139 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
140 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
141 }
142 }
143
144 if (add_to_auto_reconfigurable_params) {
145 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) { param = p.get_value<T>(); };
146 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));
147 }

150rcl_interfaces::msg::SetParametersResult Lanelet2RoutePlanning::parametersCallback(
151 const std::vector<rclcpp::Parameter>& parameters) {
152 for (const auto& param : parameters) {
153 for (auto& auto_reconfigurable_param : auto_reconfigurable_params_) {
154 if (param.get_name() == std::get<0>(auto_reconfigurable_param)) {
155 std::get<1>(auto_reconfigurable_param)(param);
156 RCLCPP_INFO(this->get_logger(), "Reconfigured parameter '%s'", param.get_name().c_str());
157 break;
158 }
159 }
160 }
161
162 // parameter-specific reconfigurations
163 for (const auto& param : parameters) {
164 if (param.get_name() == "publish_frequency") {
165 publish_timer_->cancel();
166 publish_timer_ = this->create_wall_timer(std::chrono::duration<double>(1.0 / publish_frequency_),
167 std::bind(&Lanelet2RoutePlanning::publishTimerCallback, this));
168 }
169 }
170 if (enrich_route_ahead_ego_distance_ < 0.0) {
171 enrich_route_ahead_ego_distance_ = std::numeric_limits<double>::infinity();
172 }
173 if (enrich_route_behind_ego_distance_ < 0.0) {
174 enrich_route_behind_ego_distance_ = std::numeric_limits<double>::infinity();
175 }
176 if (max_num_threads_ <= 0) {
177 max_num_threads_ = omp_get_max_threads();
178 }
179 omp_set_num_threads(max_num_threads_);
180
181 rcl_interfaces::msg::SetParametersResult result;
182 result.successful = true;
183
184 return result;
185}

199void Lanelet2RoutePlanning::setup() {
200 // map interface
201 ll2_interface_ = std::make_unique<Lanelet2MapInterface>(*this, ll2_map_server_name_);
202
203 // callback for dynamic parameter configuration
204 parameters_callback_ =
205 this->add_on_set_parameters_callback(std::bind(&Lanelet2RoutePlanning::parametersCallback, this, std::placeholders::_1));
206
207 // tf transform listener
208 tf_buffer_ = std::make_unique<tf2_ros::Buffer>(this->get_clock());
209 tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
210
211 // publishers
212 publisher_route_ = this->create_publisher<route_planning_msgs::msg::Route>("~/route", 1);
213 publish_timer_ = this->create_wall_timer(std::chrono::duration<double>(1.0 / publish_frequency_),
214 std::bind(&Lanelet2RoutePlanning::publishTimerCallback, this));
215 is_publishing_route_ = false;
216 RCLCPP_INFO(this->get_logger(), "Publishing to '%s'", publisher_route_->get_topic_name());
217
218 // subscribers
219 subscriber_ego_data_ = this->create_subscription<perception_msgs::msg::EgoData>(
220 "~/ego_data", 1, std::bind(&Lanelet2RoutePlanning::egoDataCallback, this, std::placeholders::_1));
221 RCLCPP_INFO(this->get_logger(), "Subscribed to '%s'", subscriber_ego_data_->get_topic_name());
222
223 // action server for handling action goal requests
224 action_callback_group_ = this->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
225 action_server_ = rclcpp_action::create_server<route_planning_msgs::action::PlanRoute>(
226 this, "~/plan_route",
227 std::bind(&Lanelet2RoutePlanning::actionHandleGoal, this, std::placeholders::_1, std::placeholders::_2),
228 std::bind(&Lanelet2RoutePlanning::actionHandleCancel, this, std::placeholders::_1),
229 std::bind(&Lanelet2RoutePlanning::actionHandleAccepted, this, std::placeholders::_1),
230 rcl_action_server_get_default_options(), action_callback_group_);
231 RCLCPP_INFO(this->get_logger(), "Action server started");
232}

234bool Lanelet2RoutePlanning::buildRoutingGraph() {
235 if (!this->checkMap(false)) {
236 RCLCPP_ERROR(this->get_logger(), "Cannot build routing graph, map not loaded by '%s'", ll2_map_server_name_.c_str());
237 return false;
238 }
239
240 // get map and traffic rules
241 lanelet::LaneletMapConstPtr map = ll2_interface_->getMapPtr();
242 lanelet::traffic_rules::TrafficRulesPtr traffic_rules = getTrafficRules();
243
244 // build routing graph
245 routing_graph_ = lanelet::routing::RoutingGraph::build(*map, *traffic_rules);
246 lanelet::routing::Route::Errors errors = routing_graph_->checkValidity();
247 if (!errors.empty()) {
248 RCLCPP_ERROR(this->get_logger(), "Failed to build valid routing graph");
249 for (size_t i = 0; i < errors.size(); ++i) {
250 RCLCPP_ERROR_STREAM(this->get_logger(), errors[i]);
251 }
252 return false;
253 }
254
255 RCLCPP_INFO(this->get_logger(), "Successfully built routing graph");
256 return true;

259void Lanelet2RoutePlanning::egoDataCallback(const perception_msgs::msg::EgoData::SharedPtr msg) {
260 if (!this->checkMap(false)) {
261 return;
262 }
263
264 // transform ego data to map frame
265 if (msg->header.frame_id != ll2_interface_->map_frame_id_) {
266 try {
267 latest_ego_data_ = tf_buffer_->transform(*msg, ll2_interface_->map_frame_id_, tf2::durationFromSec(transform_timeout_));
268 } catch (tf2::TransformException& ex) {
269 RCLCPP_ERROR(this->get_logger(), "Could not transform ego data from frame '%s' to frame '%s': %s",
270 msg->header.frame_id.c_str(), ll2_interface_->map_frame_id_.c_str(), ex.what());
271 }
272 } else {
273 latest_ego_data_ = *msg;
274 }
275
276 // recompute local route
277 if (is_publishing_route_) {
278 auto t0 = std::chrono::steady_clock::now();
279 this->buildEnrichedRouteMessage();
280 auto t1 = std::chrono::steady_clock::now();
281 auto dt = std::chrono::duration_cast<std::chrono::duration<double>>(t1 - t0).count();
282 RCLCPP_DEBUG(this->get_logger(), "Recomputed route (%.3fs)", dt);
283 }

292rclcpp_action::GoalResponse Lanelet2RoutePlanning::actionHandleGoal(
293 const rclcpp_action::GoalUUID& uuid, route_planning_msgs::action::PlanRoute::Goal::ConstSharedPtr goal) {
294 (void)uuid;
295 (void)goal;
296
297 const geometry_msgs::msg::PointStamped& destination = goal->destination;
298 const std::vector<geometry_msgs::msg::PointStamped>& intermediate_destinations = goal->intermediate_destinations;
299 RCLCPP_INFO(this->get_logger(), "Received request to plan route to destination (%.3f, %.3f, %.3f) in frame '%s'",
300 destination.point.x, destination.point.y, destination.point.z, destination.header.frame_id.c_str());
301
302 // check for and handle map updates
303 if (!this->checkMap(true)) {
304 RCLCPP_ERROR(this->get_logger(), "Cannot plan route, map not loaded by '%s'", ll2_map_server_name_.c_str());
305 return rclcpp_action::GoalResponse::REJECT;
306 }
307
308 // plan route
309 auto t0 = std::chrono::steady_clock::now();
310 bool success = this->planRoute(destination, intermediate_destinations);
311 auto t1 = std::chrono::steady_clock::now();
312 auto dt = std::chrono::duration_cast<std::chrono::duration<double>>(t1 - t0).count();
313 if (!success) {
314 RCLCPP_ERROR(this->get_logger(), "Failed to plan route to destination, rejecting request");
315 return rclcpp_action::GoalResponse::REJECT;
316 }
317
318 // convert route to ROS message
319 this->buildGlobalRouteMessage();
320 RCLCPP_INFO(this->get_logger(), "Successfully planned route to destination (%.3fs)", dt);
321
322 // abort current action if running
323 if (action_goal_handle_ && action_goal_handle_->is_active()) {
324 RCLCPP_WARN(this->get_logger(), "Existing action detected, aborting before accepting new goal");
325 is_publishing_route_ = false; // stop publishing route
326 action_goal_handle_->abort(action_result_);
327 action_goal_handle_.reset();
328 }
329
330 // accept action goal request
331 return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
332}

349void Lanelet2RoutePlanning::actionHandleAccepted(
350 const std::shared_ptr<rclcpp_action::ServerGoalHandle<route_planning_msgs::action::PlanRoute>> goal_handle) {
351 action_goal_handle_ = goal_handle;
352
353 // initialize feedback and result
354 action_start_time_ = this->now();
355 action_feedback_ = std::make_shared<route_planning_msgs::action::PlanRoute::Feedback>();
356 action_feedback_->distance_traveled = 0.0;
357 action_feedback_->distance_remaining = 0.0;
358 action_feedback_->distance_remaining = distanceRemaining(latest_route_msg_);
359 action_feedback_->time_traveled = rclcpp::Duration::from_seconds(0.0);
360 action_feedback_->time_remaining = rclcpp::Duration::from_seconds(estimateRemainingTime(latest_route_msg_));
361 action_result_ = std::make_shared<route_planning_msgs::action::PlanRoute::Result>();
362 action_result_->distance_traveled = 0.0;
363 action_result_->time_traveled = rclcpp::Duration::from_seconds(0.0);
364 action_result_->destination_reached = false;
365
366 // start publishing route
367 is_publishing_route_ = true;
368
369 // execute action in a separate thread to avoid blocking
370 std::thread{std::bind(&Lanelet2RoutePlanning::actionExecute, this, std::placeholders::_1), goal_handle}.detach();
371}

373void Lanelet2RoutePlanning::actionExecute(
374 const std::shared_ptr<rclcpp_action::ServerGoalHandle<route_planning_msgs::action::PlanRoute>> goal_handle) {
375 RCLCPP_INFO(this->get_logger(), "Executing action goal");
376
377 rclcpp::Rate feedback_rate(action_feedback_frequency_);
378 bool has_reached_destination = false;
379 while (goal_handle->is_executing() && !goal_handle->is_canceling() && !has_reached_destination) {
380 // update feedback and result
381 action_feedback_->distance_traveled = distanceTraveled(latest_route_msg_);
382 action_feedback_->distance_remaining = distanceRemaining(latest_route_msg_);
383 action_feedback_->time_traveled = this->now() - action_start_time_;
384 action_feedback_->time_remaining = rclcpp::Duration::from_seconds(estimateRemainingTime(latest_route_msg_));
385 action_result_->distance_traveled = action_feedback_->distance_traveled;
386 action_result_->time_traveled = action_feedback_->time_traveled;
387
388 // check if destination reached (criteria: close to destination and some distance traveled)
389 double distance_to_destination = (toEigen2d(egoPosition(latest_ego_data_)) - toEigen2d(destination_)).norm();
390 double total_route_distance = action_feedback_->distance_traveled + action_feedback_->distance_remaining;
391 bool is_close_to_destination = (distance_to_destination <= destination_distance_threshold_);
392 bool has_traveled_sufficient_distance =
393 (action_feedback_->distance_traveled >= required_traveled_distance_proportion_ * total_route_distance);
394 has_reached_destination = (is_close_to_destination && has_traveled_sufficient_distance);
395
396 // publish feedback
397 goal_handle->publish_feedback(action_feedback_);
398 if (!has_reached_destination) {
399 feedback_rate.sleep();
400 }
401 }
402
403 // prepare result
404 action_result_->distance_traveled = action_feedback_->distance_traveled + action_feedback_->distance_remaining;
405 action_result_->time_traveled = this->now() - action_start_time_;
406 action_result_->destination_reached = has_reached_destination;
407
408 // publish result
409 if (goal_handle->is_canceling()) {
410 goal_handle->canceled(action_result_);
411 RCLCPP_INFO(this->get_logger(), "Goal canceled");
412 } else if (!goal_handle->is_executing()) {
413 RCLCPP_INFO(this->get_logger(), "Goal aborted");
414 } else if (rclcpp::ok()) {
415 is_publishing_route_ = false; // stop publishing route
416 goal_handle->succeed(action_result_);
417 RCLCPP_INFO(this->get_logger(), "Goal succeeded");
418 }

421bool Lanelet2RoutePlanning::planRoute(const geometry_msgs::msg::PointStamped& destination,
422 const std::vector<geometry_msgs::msg::PointStamped>& intermediate_destinations) {
423 if (!this->checkMap(false)) {
424 RCLCPP_ERROR(this->get_logger(), "Cannot plan route, map not loaded by '%s'", ll2_map_server_name_.c_str());
425 return false;
426 }
427
428 // transform destination to map frame
429 geometry_msgs::msg::PointStamped destination_map_stamped;
430 if (destination.header.frame_id != ll2_interface_->map_frame_id_) {
431 try {
432 destination_map_stamped =
433 tf_buffer_->transform(destination, ll2_interface_->map_frame_id_, tf2::durationFromSec(transform_timeout_));
434 } catch (tf2::TransformException& ex) {
435 RCLCPP_ERROR(this->get_logger(), "Could not transform destination from frame '%s' to frame '%s': %s",
436 destination.header.frame_id.c_str(), ll2_interface_->map_frame_id_.c_str(), ex.what());
437 return false;
438 }
439 } else {
440 destination_map_stamped = destination;
441 }
442 geometry_msgs::msg::Point& destination_map = destination_map_stamped.point;
443
444 // transform intermediate destinations to map frame
445 std::vector<geometry_msgs::msg::Point> intermediate_destinations_map;
446 for (const auto& intermediate : intermediate_destinations) {
447 geometry_msgs::msg::PointStamped intermediate_map_stamped;
448 if (intermediate.header.frame_id != ll2_interface_->map_frame_id_) {
449 try {
450 intermediate_map_stamped =
451 tf_buffer_->transform(intermediate, ll2_interface_->map_frame_id_, tf2::durationFromSec(transform_timeout_));
452 } catch (tf2::TransformException& ex) {
453 RCLCPP_ERROR(this->get_logger(), "Could not transform intermediate from frame '%s' to frame '%s': %s",
454 intermediate.header.frame_id.c_str(), ll2_interface_->map_frame_id_.c_str(), ex.what());
455 return false;
456 }
457 intermediate_destinations_map.push_back(intermediate_map_stamped.point);
458 } else {
459 intermediate_destinations_map.push_back(intermediate.point);
460 }
461 }
462
463 // get map and traffic rules
464 lanelet::LaneletMapConstPtr map = ll2_interface_->getMapPtr();
465 lanelet::traffic_rules::TrafficRulesPtr traffic_rules = getTrafficRules();
466
467 // check validity of ego data
468 const double timeout_ego_data = 1.0;
469 if ((this->now() - latest_ego_data_.header.stamp).seconds() > timeout_ego_data) {
470 RCLCPP_WARN(this->get_logger(), "Ego data is outdated by %.3fs > %.3fs",
471 (this->now() - latest_ego_data_.header.stamp).seconds(), timeout_ego_data);
472 }
473 if (latest_ego_data_.header.frame_id != ll2_interface_->map_frame_id_) {
474 RCLCPP_ERROR(this->get_logger(), "Ego data frame '%s' does not match map frame '%s'",
475 latest_ego_data_.header.frame_id.c_str(), ll2_interface_->map_frame_id_.c_str());
476 return false;
477 }
478
479 // project ego position to lanelet
480 lanelet::ConstLanelet ego_ll;
481 if (auto result = laneletAtPoint(toEigen2d(egoPosition(latest_ego_data_)), map)) {
482 ego_ll = *result;
483 } else {
484 RCLCPP_ERROR(this->get_logger(), "Failed to find lanelet at ego position");
485 return false;
486 }
487 Eigen::Vector2d ego_ll_position =
488 projectPointToLineString(toEigen2d(egoPosition(latest_ego_data_)), toEigen(ego_ll.centerline2d().basicLineString()));
489
490 // project destination to lanelet
491 lanelet::ConstLanelet destination_ll;
492 if (auto result = laneletAtPoint(toEigen2d(destination_map), map)) {
493 destination_ll = *result;
494 } else {
495 RCLCPP_ERROR(this->get_logger(), "Failed to find lanelet at destination");
496 return false;
497 }
498 Eigen::Vector2d destination_ll_position =
499 projectPointToLineString(toEigen2d(destination_map), toEigen(destination_ll.centerline2d().basicLineString()));
500
501 // project intermediate destinations to lanelets
502 std::vector<lanelet::ConstLanelet> intermediate_destination_lls;
503 std::vector<geometry_msgs::msg::Point> intermediate_destinations_on_route;
504 for (const auto& intermediate : intermediate_destinations_map) {
505 lanelet::ConstLanelet intermediate_ll;
506 if (auto result = laneletAtPoint(toEigen2d(intermediate), map)) {
507 intermediate_destination_lls.push_back(*result);
508 intermediate_destinations_on_route.push_back(intermediate);
509 } else {
510 RCLCPP_WARN(this->get_logger(), "Failed to find lanelet at intermediate point (%.3f, %.3f). Skipping...", intermediate.x,
511 intermediate.y);
512 continue; // skip this intermediate if no lanelet found
513 }
514 }
515
516 // undershoot/overshoot route endpoints to enable context before start position and behind destination
517 lanelet::ConstLanelet undershot_ego_ll =
518 followLaneletsAlongRoutingGraph(routing_graph_, ego_ll, ego_ll_position, -std::abs(route_undershoot_distance_));
519 lanelet::ConstLanelet overshot_destination_ll =
520 followLaneletsAlongRoutingGraph(routing_graph_, destination_ll, destination_ll_position, route_overshoot_distance_);
521
522 // compute route from start to destination along intermediate destinations
523 std::vector<lanelet::ConstLanelet> route_lanelets = {undershot_ego_ll};
524 route_lanelets.insert(route_lanelets.end(), intermediate_destination_lls.begin(), intermediate_destination_lls.end());
525 route_lanelets.push_back(overshot_destination_ll);
526 std::optional<lanelet::routing::Route> planned_route = getRoute(routing_graph_, route_lanelets);
527
528 if (planned_route) {
529 starting_point_ = egoPosition(latest_ego_data_);
530 destination_ = destination_map;
531 intermediate_destinations_ = intermediate_destinations_on_route;
532 latest_route_ = std::move(*planned_route);
533 return true;
534 }
535 RCLCPP_ERROR(this->get_logger(), "Failed to plan route from lanelet %ld to lanelet %ld", ego_ll.id(), destination_ll.id());
536 return false;

539void Lanelet2RoutePlanning::buildGlobalRouteMessage() {
540 // create Route message
541 route_planning_msgs::msg::Route route_msg;
542 route_msg.header.stamp = latest_ego_data_.header.stamp;
543 route_msg.header.frame_id = ll2_interface_->map_frame_id_;
544 route_msg.destination = destination_;
545 route_msg.intermediate_destinations = intermediate_destinations_;
546 route_msg.route_elements = {};
547
548 // get shortest path
549 lanelet::routing::LaneletPath shortest_path = latest_route_.shortestPath();
550
551 // resample centerlines along shortest path to accumulate global reference line
552 auto resampling_result = resampleCenterlinesAlongPath(shortest_path, sampling_distance_, true);
553 std::vector<Eigen::Vector3d> shortest_path_centerline_3d = resampling_result.centerline;
554 std::vector<Eigen::Vector2d> shortest_path_centerline = to2d(shortest_path_centerline_3d);
555 latest_lanelet_idx_by_reference_line_point_idx_ = resampling_result.lanelet_idx_by_point;
556
557 // fill route message with global reference line
558 double accumulated_distance = 0;
559 for (size_t c = 0; c < shortest_path_centerline.size(); ++c) {
560 // get current, previous and next centerline point
561 const Eigen::Vector2d& point = shortest_path_centerline[c];
562 const Eigen::Vector3d& point_3d = shortest_path_centerline_3d[c];
563 const Eigen::Vector2d& prev_point = (c > 0) ? shortest_path_centerline[c - 1] : point;
564 const Eigen::Vector2d& next_point = (c < shortest_path_centerline.size() - 1) ? shortest_path_centerline[c + 1] : point;
565
566 // get lanelet corresponding to centerline point
567 const lanelet::ConstLanelet& lanelet = shortest_path[latest_lanelet_idx_by_reference_line_point_idx_[c]];
568
569 // identify lane changes based on break in equidistant centerline
570 const bool changes_lane_from_prev_point =
571 changesLaneFromPointToPoint(prev_point, point, sampling_distance_); // NOLINT(readability-suspicious-call-argument)
572 const bool changes_lane_to_next_point = changesLaneFromPointToPoint(point, next_point, sampling_distance_);
573
574 // compute orientation of centerline point
575 Eigen::Vector2d prev_point_for_orientation = changes_lane_from_prev_point ? point : prev_point;
576 Eigen::Vector2d next_point_for_orientation = changes_lane_to_next_point ? point : next_point;
577 Eigen::Vector2d orientation = tangentOfPointAlongLineString(point, prev_point_for_orientation, next_point_for_orientation);
578
579 // determine speed limit
580 uint8_t speed_limit = speedLimit(lanelet, point);
581
582 // accumulate distance
583 accumulated_distance += (point - prev_point).norm();
584
585 // create RouteElement
586 route_planning_msgs::msg::RouteElement route_element_msg = createMinimalRouteElement(
587 toRos(point_3d), toRosQuaternion(orientation), accumulated_distance, changes_lane_to_next_point, speed_limit);
588 route_msg.route_elements.push_back(route_element_msg);
589 }
590
591 // project starting point and destinations to reference line, if enabled
592 if (project_destination_to_reference_line_) {
593 starting_point_ = toRos(projectPointToLineString(toEigen(starting_point_), shortest_path_centerline_3d));
594 for (auto& destination : intermediate_destinations_) {
595 destination = toRos(projectPointToLineString(toEigen(destination), shortest_path_centerline_3d));
596 }
597 destination_ = toRos(projectPointToLineString(toEigen(destination_), shortest_path_centerline_3d));
598 route_msg.destination = destination_;
599 route_msg.intermediate_destinations = intermediate_destinations_;
600 }
601
602 // determine starting/current/destination indices in route elements
603 route_msg.starting_route_element_idx =
604 matchPointToLineString(shortest_path_centerline, toEigen2d(starting_point_), 0, true, true);
605 route_msg.current_route_element_idx = matchPointToLineString(shortest_path_centerline, toEigen2d(egoPosition(latest_ego_data_)),
606 route_msg.starting_route_element_idx, true, true);
607 route_msg.destination_route_element_idx =
608 indexOfLineStringPointClosestToPoint(shortest_path_centerline, toEigen2d(destination_), true, false);
609
610 has_enriched_route_ = false;
611 latest_route_msg_ = route_msg;
612}

614void Lanelet2RoutePlanning::buildEnrichedRouteMessage() {
615 route_planning_msgs::msg::Route route_msg = latest_route_msg_;
616 std::vector<route_planning_msgs::msg::RouteElement>& route_elements = route_msg.route_elements;
617 if (latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_.size() != route_elements.size()) {
618 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_ =
619 std::vector<std::vector<int>>(route_elements.size());
620 }
621
622 // find point of global reference line closest to and behind of ego position
623 const Eigen::Vector2d ego_position = toEigen2d(egoPosition(latest_ego_data_));
624 const std::vector<Eigen::Vector2d> reference_line = to2d(suggestedReferenceLineToEigen(route_elements));
625 size_t c_closest_point = matchPointToLineString(reference_line, ego_position, route_msg.current_route_element_idx, true, true);
626
627// loop over global reference line in parallel
628#pragma omp parallel for
629 for (size_t c = 0; c < route_elements.size(); ++c) {
630 route_planning_msgs::msg::RouteElement& route_element_msg = route_elements[c];
631 route_planning_msgs::msg::LaneElement& lane_element_msg =
632 route_element_msg.lane_elements[route_element_msg.suggested_lane_idx];
633
634 // only consider route elements within enrich_route_behind_ego_distance_ and enrich_route_ahead_ego_distance_ for local route
635 double distance_ahead = route_element_msg.s - route_elements[c_closest_point].s;
636 if (distance_ahead > enrich_route_ahead_ego_distance_ || distance_ahead < -enrich_route_behind_ego_distance_) {
637 // clear local route enriched information, if existing
638 route_element_msg = createMinimalRouteElement(lane_element_msg.reference_pose.position,
639 lane_element_msg.reference_pose.orientation, route_element_msg.s,
640 route_element_msg.will_change_suggested_lane, lane_element_msg.speed_limit);
641 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c].clear();
642 continue;
643 }
644
645 // skip recomputation of local route if already enriched
646 if (route_element_msg.is_enriched) {
647 continue;
648 }
649
650 // get current, previous and next centerline point
651 route_planning_msgs::msg::LaneElement prev_lane_element_msg, next_lane_element_msg;
652#pragma omp critical // prevent race condition when accessing prev/next suggested lane element set by other threads
653 {
654 prev_lane_element_msg =
655 (c > 0) ? route_planning_msgs::route_access::getSuggestedLaneElement(route_elements[c - 1]) : lane_element_msg;
656 next_lane_element_msg = (c < route_elements.size() - 1)
657 ? route_planning_msgs::route_access::getSuggestedLaneElement(route_elements[c + 1])
658 : lane_element_msg;
659 }
660 const Eigen::Vector2d point = toEigen2d(lane_element_msg.reference_pose.position);
661 const Eigen::Vector2d prev_point = toEigen2d(prev_lane_element_msg.reference_pose.position);
662 const Eigen::Vector2d next_point = toEigen2d(next_lane_element_msg.reference_pose.position);
663 const double point_z = lane_element_msg.reference_pose.position.z; // assuming constant z across route elements
664
665 // get lanelet corresponding to centerline point
666 lanelet::routing::LaneletPath shortest_path = latest_route_.shortestPath();
667 const lanelet::ConstLanelet& lanelet = shortest_path[latest_lanelet_idx_by_reference_line_point_idx_[c]];
668
669 // identify lane changes
670 const bool changes_lane_from_prev_point =
671 changesLaneFromPointToPoint(prev_point, point, sampling_distance_); // NOLINT(readability-suspicious-call-argument)
672 const bool changes_lane_to_next_point = changesLaneFromPointToPoint(point, next_point, sampling_distance_);
673
674 // determine neighboring points for projection
675 Eigen::Vector2d prev_point_for_projection = changes_lane_from_prev_point ? point : prev_point;
676 Eigen::Vector2d next_point_for_projection = changes_lane_to_next_point ? point : next_point;
677
678 // get adjacent lanelets
679 std::vector<lanelet::ConstLanelet> adjacent_left_lanelets = adjacentLeftOrRightLanelets(lanelet, routing_graph_, true);
680 std::vector<lanelet::ConstLanelet> adjacent_right_lanelets = adjacentLeftOrRightLanelets(lanelet, routing_graph_, false);
681 const int suggested_lane_idx = static_cast<int>(adjacent_left_lanelets.size());
682 const int n_lanes = static_cast<int>(adjacent_left_lanelets.size() + 1 + adjacent_right_lanelets.size());
683 if (latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c].empty()) {
684 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c] = std::vector<int>(n_lanes, -1);
685 }
686
687 // project centerline point to lanelet and adjacent lanelet centerlines and bounds
688 auto lanelet_projected_points =
689 projectPointToLaneletLines(point, prev_point_for_projection, next_point_for_projection,
690 std::vector<lanelet::ConstLanelet>{lanelet}, this->get_logger())[0];
691 auto adjacent_left_lanelets_projected_points = projectPointToLaneletLines(
692 point, prev_point_for_projection, next_point_for_projection, adjacent_left_lanelets, this->get_logger());
693 auto adjacent_right_lanelets_projected_points = projectPointToLaneletLines(
694 point, prev_point_for_projection, next_point_for_projection, adjacent_right_lanelets, this->get_logger());
695
696 // compute offset of lane element indices from current to next route element
697 const lanelet::ConstLanelet& lanelet_of_next_point =
698 (c < route_elements.size() - 1) ? shortest_path[latest_lanelet_idx_by_reference_line_point_idx_[c + 1]] : lanelet;
699 int following_lane_idx_offset = 0;
700 if (auto result = computeFollowingLaneIdxOffset(lanelet, lanelet_of_next_point, routing_graph_)) {
701 following_lane_idx_offset = *result;
702 } else {
703 RCLCPP_ERROR(this->get_logger(),
704 "Failed to find following lane index offset for route element %ld on lanelet %ld, assuming no offset", c,
705 lanelet.id());
706 following_lane_idx_offset = 0;
707 }
708
709 // extract drivable space
710 Eigen::Vector2d drivable_space_left, drivable_space_right;
711 std::tie(drivable_space_left, drivable_space_right) =
712 extractDrivableSpace(ll2_interface_->getMapPtr()->lineStringLayer,
713 {prev_point_for_projection, point, next_point_for_projection}, max_drivable_space_radius_);
714
715 // extract regulatory elements
716 auto regulatory_element_extraction =
717 extractRegulatoryElements(lanelet, adjacent_left_lanelets, adjacent_right_lanelets, {prev_point, point, next_point});
718
719// enrich RouteElement with local route information
720#pragma omp critical // prevent race condition when accessing prev/next suggested lane element set by other threads
721 {
722 route_element_msg.lane_elements = {};
723 route_element_msg.is_enriched = true;
724 route_element_msg.left_boundary = toRos(to3d(drivable_space_left, point_z));
725 route_element_msg.right_boundary = toRos(to3d(drivable_space_right, point_z));
726 route_element_msg.regulatory_elements = regulatory_element_extraction.regulatory_element_msgs;
727 route_element_msg.suggested_lane_idx = suggested_lane_idx;
728 route_element_msg.will_change_suggested_lane = changes_lane_to_next_point;
729 // route_element_msg.s already set in global route
730 size_t lane_element_idx = route_element_msg.lane_elements.size();
731
732 // create LaneElements for left adjacent lanes
733 for (size_t a = 0; a < adjacent_left_lanelets_projected_points.size(); ++a) {
734 route_planning_msgs::msg::LaneElement lane_element_msg;
735 lane_element_msg.reference_pose.position =
736 toRos(to3d(adjacent_left_lanelets_projected_points[a].centerline_point, point_z));
737 // lane_element_msg.reference_pose.orientation computed in postprocessRouteMessage
738 lane_element_msg.left_boundary.point = toRos(to3d(adjacent_left_lanelets_projected_points[a].left_bound_point, point_z));
739 lane_element_msg.left_boundary.type = laneBoundaryType(adjacent_left_lanelets[a].leftBound2d());
740 lane_element_msg.right_boundary.point =
741 toRos(to3d(adjacent_left_lanelets_projected_points[a].right_bound_point, point_z));
742 lane_element_msg.right_boundary.type = laneBoundaryType(adjacent_left_lanelets[a].rightBound2d());
743 lane_element_msg.speed_limit =
744 speedLimit(adjacent_left_lanelets[a], adjacent_left_lanelets_projected_points[a].centerline_point);
745 lane_element_msg.regulatory_element_idcs = regulatory_element_extraction.adjacent_left_regulatory_element_idcs[a];
746 const int computed_following_lane_idx =
747 static_cast<int>(route_element_msg.lane_elements.size()) + following_lane_idx_offset;
748 lane_element_msg.has_following_lane_idx = (computed_following_lane_idx >= 0 && computed_following_lane_idx < n_lanes);
749 if (lane_element_msg.has_following_lane_idx) {
750 lane_element_msg.following_lane_idx = computed_following_lane_idx;
751 }
752 std::tie(lane_element_msg.suggested_turn_signal,
753 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c][lane_element_idx]) =
754 suggestedTurnSignal(adjacent_left_lanelets[a], this->get_logger());
755 route_element_msg.lane_elements.push_back(lane_element_msg);
756 lane_element_idx = route_element_msg.lane_elements.size();
757 }
758
759 // create LaneElement for centerline lane
760 route_planning_msgs::msg::LaneElement centerline_lane_element_msg;
761 centerline_lane_element_msg.reference_pose.position = lane_element_msg.reference_pose.position;
762 // centerline_lane_element_msg.reference_pose.orientation computed in postprocessRouteMessage
763 centerline_lane_element_msg.left_boundary.point = toRos(to3d(lanelet_projected_points.left_bound_point, point_z));
764 centerline_lane_element_msg.left_boundary.type = laneBoundaryType(lanelet.leftBound2d());
765 centerline_lane_element_msg.right_boundary.point = toRos(to3d(lanelet_projected_points.right_bound_point, point_z));
766 centerline_lane_element_msg.right_boundary.type = laneBoundaryType(lanelet.rightBound2d());
767 centerline_lane_element_msg.speed_limit = speedLimit(lanelet, point);
768 centerline_lane_element_msg.regulatory_element_idcs = regulatory_element_extraction.regulatory_element_idcs;
769 const int computed_following_lane_idx =
770 static_cast<int>(route_element_msg.lane_elements.size()) + following_lane_idx_offset;
771 centerline_lane_element_msg.has_following_lane_idx =
772 (computed_following_lane_idx >= 0 && computed_following_lane_idx < n_lanes);
773 if (centerline_lane_element_msg.has_following_lane_idx) {
774 centerline_lane_element_msg.following_lane_idx = computed_following_lane_idx;
775 }
776 std::tie(centerline_lane_element_msg.suggested_turn_signal,
777 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c][lane_element_idx]) =
778 suggestedTurnSignal(lanelet, this->get_logger());
779 route_element_msg.lane_elements.push_back(centerline_lane_element_msg);
780 lane_element_idx = route_element_msg.lane_elements.size();
781
782 // create LaneElements for right adjacent lanes
783 for (size_t a = 0; a < adjacent_right_lanelets_projected_points.size(); ++a) {
784 route_planning_msgs::msg::LaneElement lane_element_msg;
785 lane_element_msg.reference_pose.position =
786 toRos(to3d(adjacent_right_lanelets_projected_points[a].centerline_point, point_z));
787 // lane_element_msg.reference_pose.orientation computed in postprocessRouteMessage
788 lane_element_msg.left_boundary.point = toRos(to3d(adjacent_right_lanelets_projected_points[a].left_bound_point, point_z));
789 lane_element_msg.left_boundary.type = laneBoundaryType(adjacent_right_lanelets[a].leftBound2d());
790 lane_element_msg.right_boundary.point =
791 toRos(to3d(adjacent_right_lanelets_projected_points[a].right_bound_point, point_z));
792 lane_element_msg.right_boundary.type = laneBoundaryType(adjacent_right_lanelets[a].rightBound2d());
793 lane_element_msg.speed_limit =
794 speedLimit(adjacent_right_lanelets[a], adjacent_right_lanelets_projected_points[a].centerline_point);
795 lane_element_msg.regulatory_element_idcs = regulatory_element_extraction.adjacent_right_regulatory_element_idcs[a];
796 const int computed_following_lane_idx =
797 static_cast<int>(route_element_msg.lane_elements.size()) + following_lane_idx_offset;
798 lane_element_msg.has_following_lane_idx = (computed_following_lane_idx >= 0 && computed_following_lane_idx < n_lanes);
799 if (lane_element_msg.has_following_lane_idx) {
800 lane_element_msg.following_lane_idx = computed_following_lane_idx;
801 }
802 std::tie(lane_element_msg.suggested_turn_signal,
803 latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_[c][lane_element_idx]) =
804 suggestedTurnSignal(adjacent_right_lanelets[a], this->get_logger());
805 route_element_msg.lane_elements.push_back(lane_element_msg);
806 lane_element_idx = route_element_msg.lane_elements.size();
807 }
808 }
809 }
810
811 // split in traveled and remaining route elements
812 route_msg.current_route_element_idx = c_closest_point;
813 route_msg.header.stamp = latest_ego_data_.header.stamp;
814
815 // postprocess route message
816 postprocessRouteMessage(route_msg, latest_suggested_turn_signal_distance_ahead_by_route_element_by_lane_element_);
817
818 // save as latest route message
819 latest_route_msg_ = route_msg;
820 has_enriched_route_ = true;
821}
