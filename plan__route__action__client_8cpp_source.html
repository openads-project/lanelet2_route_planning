23std::optional<std::vector<std::pair<double, double>>> parseWaypoints(const std::vector<std::string>& waypoints_param) {
24 std::vector<std::pair<double, double>> waypoints;
25 for (const auto& waypoint : waypoints_param) {
26 size_t comma_pos = waypoint.find(',');
27 if (comma_pos != std::string::npos) {
28 try {
29 double lat = std::stod(waypoint.substr(0, comma_pos));
30 double lon = std::stod(waypoint.substr(comma_pos + 1));
31 waypoints.emplace_back(lat, lon);
32 } catch (const std::invalid_argument& e) {
33 return std::nullopt;
34 } catch (const std::out_of_range& e) {
35 return std::nullopt;
36 }
37 } else {
38 return std::nullopt;
39 }
40 }
41 return waypoints;
42}

60void PlanRouteActionClient::declareAndLoadParameter(const std::string& name,
61 T& param,
62 const std::string& description,
63 const bool add_to_auto_reconfigurable_params,
64 const bool is_required,
65 const bool read_only,
66 const std::optional<double>& from_value,
67 const std::optional<double>& to_value,
68 const std::optional<double>& step_value,
69 const std::string& additional_constraints) {
70 rcl_interfaces::msg::ParameterDescriptor param_desc;
71 param_desc.description = description;
72 param_desc.additional_constraints = additional_constraints;
73 param_desc.read_only = read_only;
74
75 auto type = rclcpp::ParameterValue(param).get_type();
76
77 if (from_value.has_value() && to_value.has_value()) {
78 if constexpr (std::is_integral_v<T>) {
79 rcl_interfaces::msg::IntegerRange range;
80 T step = static_cast<T>(step_value.has_value() ? step_value.value() : 1);
81 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value())).set__step(step);
82 param_desc.integer_range = {range};
83 } else if constexpr (std::is_floating_point_v<T>) {
84 rcl_interfaces::msg::FloatingPointRange range;
85 T step = static_cast<T>(step_value.has_value() ? step_value.value() : 1.0);
86 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value())).set__step(step);
87 param_desc.floating_point_range = {range};
88 } else {
89 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range", name.c_str());
90 }
91 }
92
93 this->declare_parameter(name, type, param_desc);
94
95 try {
96 param = this->get_parameter(name).get_value<T>();
97 std::stringstream ss;
98 ss << "Loaded parameter '" << name << "': ";
99 if constexpr (is_vector_v<T>) {
100 ss << "[";
101 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "]");
102 } else {
103 ss << param;
104 }
105 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
106 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
107 if (is_required) {
108 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");
109 exit(EXIT_FAILURE);
110 } else {
111 std::stringstream ss;
112 ss << "Missing parameter '" << name << "', using default value: ";
113 if constexpr (is_vector_v<T>) {
114 ss << "[";
115 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "]");
116 } else {
117 ss << param;
118 }
119 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
120 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
121 }
122 }
123
124 if (add_to_auto_reconfigurable_params) {
125 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) { param = p.get_value<T>(); };
126 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));
127 }

130rcl_interfaces::msg::SetParametersResult PlanRouteActionClient::parametersCallback(
131 const std::vector<rclcpp::Parameter>& parameters) {
132 for (const auto& param : parameters) {
133 for (auto& auto_reconfigurable_param : auto_reconfigurable_params_) {
134 if (param.get_name() == std::get<0>(auto_reconfigurable_param)) {
135 std::get<1>(auto_reconfigurable_param)(param);
136 RCLCPP_INFO(this->get_logger(), "Reconfigured parameter '%s'", param.get_name().c_str());
137 break;
138 }
139 }
140
141 // handle waypoints
142 if (param.get_name() == "waypoints") {
143 auto parsed_waypoints = parseWaypoints(waypoints_param_);
144 if (parsed_waypoints) {
145 waypoints_ = *parsed_waypoints;
146 } else {
147 std::stringstream ss;
148 ss << "Failed to parse parameter 'waypoints': [";
149 for (const auto& waypoint : waypoints_param_) {
150 ss << waypoint << (&waypoint != &waypoints_param_.back() ? ", " : "]");
151 }
152 RCLCPP_ERROR(this->get_logger(), "%s", ss.str().c_str());
153 }
154 }
155
156 // handle cancel_route
157 if (param.get_name() == "cancel_route") {
158 if (cancel_route_) {
159 if (action_client_->wait_for_action_server(std::chrono::duration<double>(0.1))) {
160 RCLCPP_INFO(this->get_logger(), "Cancelling route");
161 action_client_->async_cancel_all_goals();
162 } else {
163 RCLCPP_WARN(this->get_logger(), "Action server not available, cannot cancel route");
164 }
165 }
166 }
167 }
168
169 rcl_interfaces::msg::SetParametersResult result;
170 result.successful = true;
171
172 return result;
173}

193 if (parsed_waypoints) {

199 ss << waypoint << (&waypoint != &waypoints_param_.back() ? ", " : "]");
200 }
201 RCLCPP_ERROR(this->get_logger(), "%s", ss.str().c_str());
202 }

204 // set up auto-planning timer

215void PlanRouteActionClient::autoPlanningTimerCallback() {
216 if (enable_random_destination_ && (enable_continuous_planning_ || !has_completed_one_goal_)) {
217 this->planToRandomDestination();
218 } else if (!waypoints_.empty()) {
219 if (next_waypoint_idx_ >= waypoints_.size() && enable_continuous_planning_) {
220 next_waypoint_idx_ = 0; // loop waypoints, if continuous planning is enabled
221 }
222 if (next_waypoint_idx_ < waypoints_.size()) {
223 this->planToNextWaypoint();
224 }
225 } else {
226 RCLCPP_DEBUG(this->get_logger(), "Nothing to plan, waiting for waypoints or random destination");
227 }

230void PlanRouteActionClient::planToNextWaypoint() {
231 // check if waypoint is valid
232 if (next_waypoint_idx_ >= waypoints_.size()) {
233 RCLCPP_ERROR(this->get_logger(), "Waypoint index %ld out of bounds (%ld), skipping", next_waypoint_idx_, waypoints_.size());
234 return;
235 }
236 RCLCPP_INFO(this->get_logger(), "Planning route to next waypoint (%.6f, %.6f)", waypoints_[next_waypoint_idx_].first,
237 waypoints_[next_waypoint_idx_].second);
238
239 // check if map is loaded
240 if (!ll2_interface_->map_loaded_) {
241 RCLCPP_ERROR(this->get_logger(), "Map not loaded, cannot generate waypoint");
242 return;
243 }
244
245 // generate goal pose from waypoint
246 auto goal_pose = std::make_shared<geometry_msgs::msg::PoseStamped>();
247 auto ll2_projector = ll2_interface_->getProjectorPtr();
248 if (ll2_projector) {
249 lanelet::GPSPoint gps_waypoint;
250 gps_waypoint.lat = waypoints_[next_waypoint_idx_].first;
251 gps_waypoint.lon = waypoints_[next_waypoint_idx_].second;
252 lanelet::BasicPoint3d map_waypoint = ll2_projector->forward(gps_waypoint);
253 goal_pose->pose.position.x = map_waypoint.x();
254 goal_pose->pose.position.y = map_waypoint.y();
255 goal_pose->pose.position.z = 0.0;
256 goal_pose->header.frame_id = ll2_interface_->map_frame_id_;
257 goal_pose->header.stamp = this->now();
258 }
259
260 // send goal
261 if (!goal_pose->header.frame_id.empty()) {
262 RCLCPP_INFO(this->get_logger(), "Generated waypoint goal pose (%.3f, %.3f, %.3f) in frame '%s'", goal_pose->pose.position.x,
263 goal_pose->pose.position.y, goal_pose->pose.position.z, goal_pose->header.frame_id.c_str());
264 auto_planning_timer_->cancel(); // cancel auto-planning timer until goal completion
265 next_waypoint_idx_++;
266 this->sendGoal(goal_pose);
267 } else {
268 RCLCPP_ERROR(this->get_logger(), "Failed to generate waypoint goal pose");
269 }
270}

272void PlanRouteActionClient::planToRandomDestination() {
273 RCLCPP_INFO(this->get_logger(), "Planning route to random destination");
274
275 // check if map is loaded
276 if (!ll2_interface_->map_loaded_) {
277 RCLCPP_ERROR(this->get_logger(), "Map not loaded, cannot generate a random destination");
278 return;
279 }
280
281 // generate random goal pose by sampling a random lanelet
282 auto goal_pose = std::make_shared<geometry_msgs::msg::PoseStamped>();
283 lanelet::LaneletMapConstPtr map = ll2_interface_->getMapPtr();
284 if (!map->laneletLayer.empty()) {
285 const auto lanelet_count = static_cast<int>(map->laneletLayer.size());
286 const auto random_lanelet_idx =
287 static_cast<std::iterator_traits<decltype(map->laneletLayer.begin())>::difference_type>(std::rand() % lanelet_count);
288 auto random_lanelet = *std::next(map->laneletLayer.begin(), random_lanelet_idx);
289 auto centerline = random_lanelet.centerline();
290 if (!centerline.empty()) {
291 auto point = centerline.back();
292 goal_pose->pose.position.x = point.x();
293 goal_pose->pose.position.y = point.y();
294 goal_pose->pose.position.z = point.z();
295 if (centerline.size() > 1) {
296 auto heading =
297 std::atan2(point.y() - centerline[centerline.size() - 2].y(), point.x() - centerline[centerline.size() - 2].x());
298 tf2::Quaternion q;
299 q.setRPY(0, 0, heading);
300 goal_pose->pose.orientation = tf2::toMsg(q);
301 }
302 goal_pose->header.frame_id = ll2_interface_->map_frame_id_;
303 goal_pose->header.stamp = this->now();
304 }
305 }
306
307 // send goal
308 if (!goal_pose->header.frame_id.empty()) {
309 RCLCPP_INFO(this->get_logger(), "Generated random goal pose (%.3f, %.3f, %.3f) in frame '%s'", goal_pose->pose.position.x,
310 goal_pose->pose.position.y, goal_pose->pose.position.z, goal_pose->header.frame_id.c_str());
311 auto_planning_timer_->cancel(); // cancel auto-planning timer until goal completion
312 this->sendGoal(goal_pose);
313 } else {
314 RCLCPP_ERROR(this->get_logger(), "Failed to generate random goal pose");
315 }
316}

318void PlanRouteActionClient::sendGoal(const geometry_msgs::msg::PoseStamped::SharedPtr msg) {
319 RCLCPP_INFO(this->get_logger(), "Requesting to plan route to destination (%.3f, %.3f, %.3f) in frame '%s'",
320 msg->pose.position.x, msg->pose.position.y, msg->pose.position.z, msg->header.frame_id.c_str());
321
322 // check if action server is available
323 if (!action_client_->wait_for_action_server(std::chrono::duration<double>(0.1))) {
324 RCLCPP_ERROR(this->get_logger(), "Action server not available, aborting");
325 auto_planning_timer_->reset(); // restart auto-planning timer
326 return;
327 }
328
329 // build goal
330 auto goal = PlanRoute::Goal();
331 goal.destination = geometry_msgs::msg::PointStamped();
332 goal.destination.header = msg->header;
333 goal.destination.point = msg->pose.position;
334
335 // send goal
336 auto send_goal_options = rclcpp_action::Client<PlanRoute>::SendGoalOptions();
337 send_goal_options.goal_response_callback = std::bind(&PlanRouteActionClient::goalResponseCallback, this, std::placeholders::_1);
338 send_goal_options.feedback_callback =
339 std::bind(&PlanRouteActionClient::feedbackCallback, this, std::placeholders::_1, std::placeholders::_2);
340 send_goal_options.result_callback = std::bind(&PlanRouteActionClient::resultCallback, this, std::placeholders::_1);
341 goal_handle_future_ = action_client_->async_send_goal(goal, send_goal_options);
342 RCLCPP_INFO(this->get_logger(), "Goal sent");
343}

354void PlanRouteActionClient::feedbackCallback(GoalHandlePlanRoute::SharedPtr goal_handle,
355 const std::shared_ptr<const PlanRoute::Feedback> feedback) {
356 (void)goal_handle;
357
358 const double distance_traveled = feedback->distance_traveled;
359 const double distance_total = feedback->distance_remaining + feedback->distance_traveled;
360 rclcpp::Duration time_traveled(feedback->time_traveled.sec, feedback->time_traveled.nanosec);

362 rclcpp::Duration time_total = time_traveled + time_remaining;

367void PlanRouteActionClient::resultCallback(const GoalHandlePlanRoute::WrappedResult& result) {
368 const double distance_traveled = result.result->distance_traveled;
369 const builtin_interfaces::msg::Duration& time_traveled = result.result->time_traveled;
370 if (result.code == rclcpp_action::ResultCode::SUCCEEDED) {
371 if (result.result->destination_reached) {
372 RCLCPP_INFO(this->get_logger(), "Goal succeeded: destination reached after %.2fm and %ds", distance_traveled,
373 time_traveled.sec);
374 } else {
375 RCLCPP_WARN(this->get_logger(), "Goal succeeded, but destination not reached after %.2fm and %ds", distance_traveled,
376 time_traveled.sec);
377 }
378 } else if (result.code == rclcpp_action::ResultCode::CANCELED) {
379 RCLCPP_WARN(this->get_logger(), "Goal canceled: traveled %.2fm and %ds", distance_traveled, time_traveled.sec);
380 } else if (result.code == rclcpp_action::ResultCode::ABORTED) {
381 RCLCPP_ERROR(this->get_logger(), "Goal aborted: traveled %.2fm and %ds", distance_traveled, time_traveled.sec);
382 } else {
383 RCLCPP_ERROR(this->get_logger(), "Goal finished with unknown result code: %d", static_cast<int>(result.code));
384 }
385
386 auto_planning_timer_->reset(); // restart auto-planning timer
387 has_completed_one_goal_ = true;
388}
